function lookUpProfile(name, prop){
    for (var i = 0; i < contacts.length; i++) {
        if(contacts[i].firstName == name) {
            if(contacts[i].hasOwnProperty(prop)) {
                return contacts[i][prop];
            } else {
                return "No such property";
            }
        }
    } return "No such contact";
}

def convert_to_ordinal(number)
    if number % 100 >= 11 && number % 100 <= 13
        return 'th'
    elsif number % 10 == 1
        return 'st'
    elsif number % 10 == 2
        return 'nd'
    elsif number % 10 == 3
        return 'rd'
    else
        return 'th'
    end
end

Foobar

This ruby program is an adaptation of a classsic technical interview problem which displays sequences of the Foobar pattern.

> 
def foobar_series(items)

    array = []
    items.times do |number|
        if ((number + 1) % 3 == 0) && ! ((number + 1) % 5 == 0)
            array.push("Foo")
        elsif ! ((number + 1) % 3 == 0) && ((number + 1) % 5 == 0)
            array.push("Bar")
        elsif ((number + 1) % 3 == 0) && ((number + 1) % 5 == 0)
            array.push("Foobar")
        else
            array.push(number + 1)
        end
    end
    
    return array
end
